\section*{Phase 1: The Number Sets (The Domain of Energy)}
\begin{center}
\begin{tabular}{|c|c|p{5.5cm}|p{4cm}|c|}
\hline
\textbf{Symbol} & \textbf{Name} & \textbf{Definition / Range} & \textbf{Description} & \textbf{Example (Input 7)} \\
\hline
$\mathbb{N}$ & Natural Numbers & $\{1, 2, 3, \dots, \infty\}$ & Pure counting numbers. & $7 \in \mathbb{N}$ \\
\hline
$\mathbb{W}$ & Whole Numbers & $\{0, 1, 2, 3, \dots, \infty\}$ & Natural numbers plus zero. & $7 \in \mathbb{W}$ \\
\hline
$\mathbb{Z}$ & Integers & $\{\dots, -2, -1, 0, 1, 2, \dots\}$ & All whole numbers and their negatives. & $7 \in \mathbb{Z}$ \\
\hline
$\mathbb{Q}$ & Rational Numbers & $\{p/q \mid p, q \in \mathbb{Z}, q \neq 0\}$ & Can be written as $p/q$. Terminating or repeating decimals. & $7 = 7/1$. $7 \in \mathbb{Q}$ \\
\hline
$\mathbb{I}$ & Irrational Numbers & Non-repeating, infinite decimals. & Cannot be written as a simple fraction. & $7$ is NOT irrational, but $7+\pi$ IS. \\
\hline
$\mathbb{R}$ & Real Numbers & $\mathbb{Q} \cup \mathbb{I}$ (All $\mathbb{Q}$ and $\mathbb{I}$) & The entire continuous number line. & $7 \in \mathbb{R}$ \\
\hline
$\mathbb{C}$ & Complex Numbers & $\{a + bi \mid a, b \in \mathbb{R}\}$ & Real part ($a$) and Imaginary part ($b$). $i = \sqrt{-1}$. & $7 = 7 + 0i$. $7 \in \mathbb{C}$ \\
\hline
\end{tabular}
\end{center}
